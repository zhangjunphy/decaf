   25 -- operators
   26 data RelOp
   27   = LessThan
   28   | GreaterThan
   29   | LessEqual
   30   | GreaterEqual
   31   deriving Eq
   32 instance Show RelOp where
   33   show LessThan = "<"
   34   show GreaterThan = ">"
   35   show LessEqual = "<="
   36   show GreaterEqual = ">="
   37 
   38 data ArithOp
   39   = Plus
   40   | Minus
   41   | Multiply
   42   | Division
   43   | Modulo
   44   deriving Eq
   45 instance Show ArithOp where
   46   show Plus = "+"
   47   show Minus = "-"
   48   show Multiply = "*"
   49   show Division = "/"
   50   show Modulo = "%"
   51 
   52 data EqOp
   53   = Equal
   54   | NotEqual
   55   deriving Eq
   56 instance Show EqOp where
   57   show Equal = "=="
   58   show NotEqual = "!="
   59 
   60 data CondOp
   61   = OR
   62   | AND
   63   deriving Eq
   64 instance Show CondOp where
   65   show OR = "||"
   66   show AND = "&&"
   67 
   68 data NegOp
   69   = Neg
   70   deriving Eq
   71 instance Show NegOp where
   72   show Neg = "-"
   73 
   74 data NotOp
   75   = Not
   76   deriving Eq
   77 instance Show NotOp where
   78   show Not = "!"
   79 
   80 data ChoiceOp
   81   = Choice
   82   deriving (Show, Eq)
   83 
   84 data AssignOp
   85   = EqlAssign
   86   | IncAssign
   87   | DecAssign
   88   | PlusPlus
   89   | MinusMinus
   90   deriving Eq
   91 instance Show AssignOp where
   92   show EqlAssign = "="
   93   show IncAssign = "+="
   94   show DecAssign = "-="
   95   show PlusPlus = "++"
   96   show MinusMinus = "--"
   97 
   98 data Type
   99   = Void
  100   | IntType
  101   | BoolType
  102   | StringType
  103   | ArrayType !Type !Int64
  104   | Ptr !Type
  105   deriving (Eq)
  106 instance Show Type where
  107   show Void = "void"
  108   show IntType = "int"
  109   show BoolType = "bool"
  110   show StringType = "string"
  111   show (ArrayType tpe size) = formatToString (shown % "x" % shown) size tpe
  112   show (Ptr tpe) = formatToString ("ptr" %+ shown) tpe
  113 
  114 dataSize :: Type -> Maybe Int64
  115 dataSize Void = Nothing
  116 dataSize IntType = Just 8
  117 dataSize BoolType = Just 1
  118 dataSize StringType = Just 8
  119 dataSize (ArrayType tpe size) = dataSize tpe <&> \s -> s * size
  120 dataSize (Ptr _) = Just 8
  121 
  122 parseArithOp :: Text -> ArithOp
  123 parseArithOp op = case op of
  124   "+" -> Plus
  125   "-" -> Minus
  126   "*" -> Multiply
  127   "/" -> Division
  128   "%" -> Modulo
  129 
  130 parseRelOp :: Text -> RelOp
  131 parseRelOp op = case op of
  132   "<" -> LessThan
  133   ">" -> GreaterThan
  134   "<=" -> LessEqual
  135   ">=" -> GreaterEqual
  136 
  137 parseEqOp :: Text -> EqOp
  138 parseEqOp op = case op of
  139   "==" -> Equal
  140   "!=" -> NotEqual
  141 
  142 parseCondOp :: Text -> CondOp
  143 parseCondOp op = case op of
  144   "||" -> OR
  145   "&&" -> AND
  146 
  147 parseNegOp :: Text -> NegOp
  148 parseNegOp op = case op of
  149   "-" -> Neg
  150 
  151 parseNotOp :: Text -> NotOp
  152 parseNotOp op = case op of
  153   "!" -> Not
  154 
  155 parseAssignOp :: Text -> AssignOp
  156 parseAssignOp s = case s of
  157   "+=" -> IncAssign
  158   "-=" -> DecAssign
  159   "=" -> EqlAssign
  160   "++" -> PlusPlus
  161   "--" -> MinusMinus

  163 -- auxiliary data types
  164 data Location = Location
  165   { name :: Name,
  166     idx :: Maybe Expr,
  167     variableDef :: Either Argument FieldDecl,
  168     tpe :: Type,
  169     loc :: SL.Range
  170   } deriving (Generic)
  171 
  172 typeOfDef :: Either Argument FieldDecl -> Type
  173 typeOfDef (Left Argument{tpe=tpe}) = tpe
  174 typeOfDef (Right FieldDecl{tpe=tpe}) = tpe
  175 
  176 instance Show Location where
  177   show Location{name=nm, idx=idx} = printf "Location {name=%s, idx=%s}" nm (show idx)
  178 
  179 data Assignment = Assignment
  180   { location :: Location,
  181     op :: AssignOp,
  182     expr :: Maybe Expr,
  183     loc :: SL.Range
  184   } deriving (Generic, Show)
  185 
  186 data MethodCall = MethodCall
  187   { name :: Name,
  188     args :: [Expr],
  189     loc :: SL.Range
  190   } deriving (Generic, Show)

  204 data FieldDecl = FieldDecl

  232 data Statement = Statement
  233   { statement_ :: Statement_
  234   , loc :: SL.Range
  235   } deriving (Generic, Show)
  236 data Statement_
  237   = AssignStmt {assign :: Assignment}
  238   | IfStmt {pred :: Expr, ifBlock :: Block, elseBlock :: Maybe Block}
  239   | ForStmt
  240       { counter :: Name,
  241         initCounter :: Expr,
  242         pred :: Expr,
  243         update :: Assignment,
  244         block :: Block
  245       }
  246   | WhileStmt {pred :: Expr, block :: Block}
  247   | ReturnStmt {expr :: Maybe Expr}
  248   | MethodCallStmt {methodCall :: MethodCall}
  249   | BreakStmt
  250   | ContinueStmt
  251   deriving (Generic, Show)
  252 
  253 data Expr = Expr
  254   { expr_ :: Expr_
  255   , tpe :: Type
  256   , loc :: SL.Range
  257   } deriving (Generic, Show)
  258 
  259 data Expr_
  260   = LocationExpr {location :: Location}
  261   | MethodCallExpr {methodCall :: MethodCall}
  262   | ExternCallExpr {name :: Name, args :: [Expr]}
  263   | IntLiteralExpr {intVal :: Int64}
  264   | BoolLiteralExpr {boolVal :: Bool}
  265   | CharLiteralExpr {charVal :: Char}
  266   | StringLiteralExpr {strVal :: Text}
  267   | ArithOpExpr {arithOp :: ArithOp, lhs :: Expr, rhs :: Expr}
  268   | RelOpExpr {relOp :: RelOp, lhs :: Expr, rhs :: Expr}
  269   | CondOpExpr {condOp :: CondOp, lhs :: Expr, rhs :: Expr}
  270   | EqOpExpr {eqOp :: EqOp, lhs :: Expr, rhs :: Expr}
  271   | NegOpExpr {negOp :: NegOp, expr :: Expr}
  272   | NotOpExpr {notOp :: NotOp, expr :: Expr}
  273   | ChoiceOpExpr {choiceOp :: ChoiceOp, expr1 :: Expr, expr2 :: Expr, expr3 :: Expr}
  274   | LengthExpr {name :: Name}
  275   deriving (Generic, Show)
  276 
  277 data Typed a = Typed {ele :: a, tpe :: Type}
  278   deriving (Generic, Show)
  279 
  280 data Block = Block
  281   { vars :: [FieldDecl],
  282     stmts :: [Statement],
  283     blockID :: ScopeID
  284   }
  285   deriving (Generic, Show)
