   28 -- Produce error tokens for later use
   29 catchErrors :: Alex (SL.Located Token) -> Alex (SL.Located Token)
   30 catchErrors (Alex al) =
   31   Alex
   32     ( \s -> case al s of
   33         Right (s'@AlexState {alex_ust = ust}, t@(SL.LocatedAt _ EOF)) -> eofCheck s' t
   34         Right (s', x) -> Right (s', x)
   35         Left message -> Left message
   36     )
   37   where
   38     eofCheck s@AlexState {alex_ust = ust} tok@(SL.LocatedAt loc _) =
   39       case ust of
   40         val
   41           | lexerStringState ust -> Right (s, SL.LocatedAt loc $ Error "string not closed at EOF")
   42           | lexerCharState ust -> Right (s, SL.LocatedAt loc $ Error "char not closed at EOF")
   43           | lexerCommentDepth ust > 0 -> Right (s, SL.LocatedAt loc $ Error "comment not closed at EOF")
   44           | otherwise -> Right (s, tok)
   45 
   46 scan :: ByteString -> [Either String (SL.Located Token)]
   47 scan str =
   48   let loop = do
   49         tokOrError <- catchErrors alexMonadScan
   50         case tokOrError of
   51           t@(SL.LocatedAt _ (Error m)) ->
   52             do
   53               toks <- loop
   54               return (Right t : toks)
   55           t@(SL.LocatedAt _ tok) ->
   56             if tok == EOF
   57               then return []
   58               else do
   59                 toks <- loop
   60                 return (Right t : toks)
   61    in case runAlex str loop of
   62         Left m -> [Left m]
   63         Right toks -> toks
   64 
   65 formatTokenOrError :: Either String (SL.Located Token) -> Either String String
   66 formatTokenOrError (Left err) = Left err
   67 formatTokenOrError (Right (SL.LocatedAt (SL.Range (SL.Posn _ line _) _) tok)) =
   68   Right $ unwords [show line, show tok]
