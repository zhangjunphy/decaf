   12 {-# LANGUAGE DuplicateRecordFields #-}
   13 {-# LANGUAGE GeneralizedNewtypeDeriving #-}
   14 {-# LANGUAGE OverloadedStrings #-}
   15 
   16 module Graph where
   17 
   18 import Control.Monad
   19 import Data.Functor
   20 import Data.Map (Map)
   21 import qualified Data.Map.Strict as Map
   22 import Data.Set (Set)
   23 import qualified Data.Set as Set
   24 
   25 newtype Node = Node Int
   26   deriving (Eq, Ord, Show)
   27 
   28 data Graph nd ed = Graph
   29   { nodes :: Map Node nd,
   30     edges :: Map Node [(Node, ed)],
   31     start :: Maybe Node,
   32     nextNodeIdx :: Int
   33   }
   34   deriving (Show)
   35 
   36 empty :: Graph nd ed
   37 empty = Graph Map.empty Map.empty Nothing 0
   38 
   39 addNode :: nd -> Graph nd ed -> (Node, Graph nd ed)
   40 addNode dt cfg =
   41   let n = Node (nextNodeIdx cfg)
   42    in (n, cfg {nodes = Map.insert n dt (nodes cfg), nextNodeIdx = nextNodeIdx cfg + 1})
   43 
   44 addEdge :: Node -> Node -> ed -> Graph nd ed -> Graph nd ed
   45 addEdge start end dt cfg =
   46   let edges' = Map.insertWith (++) start [(end, dt)] (edges cfg)
   47    in cfg {edges = edges'}
   48 
   49 deleteEdge :: Node -> Node -> Graph nd ed -> Graph nd ed
   50 deleteEdge start end cfg =
   51   let edges' = Map.update (Just <$> filter (\(e, dt) -> e /= end)) start (edges cfg)
   52    in cfg {edges = edges'}
   53 
   54 deleteNode :: Node -> Graph nd ed -> Graph nd ed
   55 deleteNode node cfg =
   56   let nodes' = Map.delete node (nodes cfg)
   57       edges' = Map.filterWithKey (\k v -> k /= node) (edges cfg)
   58       edges'' = Map.mapMaybe (Just <$> filter (\(end, dt) -> end /= node)) edges'
   59    in cfg {nodes = nodes'}
   60 
   61 outBound :: Node -> Graph nd ed -> [(Node, ed)]
   62 outBound node g = concat $ Map.lookup node (edges g)
   63 
   64 inBound :: Node -> Graph nd ed -> [(Node, ed)]
   65 inBound node cfg =
   66   let edges' = Map.mapMaybe (Just <$> filter (\(end, _) -> end == node)) (edges cfg)
   67       nodes' =
   68         concat (Map.assocs edges' <&> \(src, dsts) -> zip (repeat src) dsts)
   69           <&> \(src, (_, ed)) -> (src, ed)
   70    in nodes'
