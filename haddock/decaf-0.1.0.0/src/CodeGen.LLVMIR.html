-- This should be a straightforward translation from CFG into LLVM IR

module CodeGen.LLVMIR where

import Data.Text (Text)

data Module = Module 
  { Module -> [Global]
globals :: ![Global]
  , Module -> [Function]
functions :: ![Function]
  }

data Global = Global
  { Global -> Text
name :: !Text
  , Global -> Type
tpe :: !Type
  }

data Function = Function
  { Function -> Text
name :: !Text
  , Function -> [Argument]
arguments :: ![Argument]
  , Function -> [BasicBlock]
basicBlocks :: ![BasicBlock]
  }

data Argument = Argument
  { Argument -> Text
name :: !Text
  , Argument -> Type
tpe :: !Type
  }

data Type =
  VoidType
  | IntType !Int
  | PointerType !Type
  | ArrayType !Type !Int

data BasicBlock = BasicBlock
  { BasicBlock -> Text
label :: !Text
  , BasicBlock -> [Instruction]
instructions :: ![Instruction]
  }

data Instruction = 
  Terminator !TermInst
  | Binary !BinaryInst
  | MemAccess !MemAccInst
  | Phi
  | Select
  | Call

data TermInst =
  Ret
  | Br

data BinaryInst =
  Add
  | Sub
  | Mul
  | SDiv

data MemAccInst =
  Alloca
  | Load
  | Store
